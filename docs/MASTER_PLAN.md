### 2.2 IP 기반 자동 지역 감지

```typescript
// middleware.ts
import { NextRequest, NextResponse } from 'next/server';

export async function middleware(request: NextRequest) {
  // 1. 사용자 IP 추출
  const ip = request.headers.get('x-forwarded-for')?.split(',')[0] ||
             request.headers.get('x-real-ip') ||
             '127.0.0.1';

  // 2. GeoIP API로 지역 감지
  const geo = await fetch(`https://ip-api.com/json/${ip}?lang=ko`);
  const { city, regionName } = await geo.json();

  // 3. 지역 코드 매핑
  const regionCode = mapCityToRegionCode(city, regionName);
  // 예: '나주' → 'naju', '광주' → 'gwangju'

  // 4. 쿠키에 지역 정보 저장 (사용자가 변경 가능)
  const response = NextResponse.next();
  response.cookies.set('user_region', regionCode, {
    maxAge: 60 * 60 * 24 * 7  // 7일
  });

  return response;
}

// 지역 코드 매핑 함수
function mapCityToRegionCode(city: string, regionName: string): string {
  const regionMap: Record<string, string> = {
    '나주': 'naju',
    '나주시': 'naju',
    '목포': 'mokpo',
    '목포시': 'mokpo',
    '순천': 'suncheon',
    '순천시': 'suncheon',
    '여수': 'yeosu',
    '여수시': 'yeosu',
    '광양': 'gwangyang',
    '광양시': 'gwangyang',
    '광주': 'gwangju',
    '광주광역시': 'gwangju',
    // ... 전남 22개 시군 + 광주 5개 구
  };

  return regionMap[city] || regionMap[regionName] || 'gwangju';
}
```

### 3.2 신규 테이블 (Phase 2)

```sql
-- 관광명소 테이블
CREATE TABLE places (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,          -- 나주영상테마파크
  region VARCHAR(50) NOT NULL,          -- naju
  category VARCHAR(50),                 -- 관광지, 맛집, 문화재
  address TEXT,
  lat DECIMAL(10, 8),
  lng DECIMAL(11, 8),
  description TEXT,
  thumbnail_url TEXT,
  naver_map_url TEXT,                   -- 네비 링크
  kakao_map_url TEXT,
  opening_hours TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 지역 행사 테이블
CREATE TABLE events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title VARCHAR(200) NOT NULL,
  region VARCHAR(50) NOT NULL,
  event_date DATE NOT NULL,
  end_date DATE,
  location TEXT,
  description TEXT,
  thumbnail_url TEXT,
  source_url TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 지역 날씨 캐시 테이블 (성능 최적화)
CREATE TABLE weather_cache (
  region VARCHAR(50) PRIMARY KEY,
  weather_data JSONB,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```
